  }

#### Import the SCSS:
<Source
  code={

  }
/>

#### SCSS customization variables with default values:
<Source
  language="css"
  code={
    dedent`
      $kws-theme-colors: $colors;
      $kws-toggle-off-bg-color: $background;
      $kws-toggle-handle-bg-color: $white;
      $kws-toggle-track-bg-color: $text;
      $kws-toggle-track-bg-color-invert: findColorInvert($kws-toggle-track-bg-color);
      $kws-toggle-border-color: $input-border-color;
    `
  }
/>
