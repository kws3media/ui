import { Story, Preview, Meta, ArgsTable, Source, Description, Canvas  } from '@storybook/addon-docs';
import dedent from 'ts-dedent';
import { ConfirmButton as Button } from '@kws3/ui';
import { action } from '@storybook/addon-actions';

export let args={
  color : "primary",
  disabled : false,
  size : "",
  class : "",
  icon_only : false,
  text : "Do it?",
  icon : "minus-circle",
  doing_icon : "hourglass",
  done_icon : "check",
  doing_text : "Doing",
  done_text : "Deleted"
};

export function confirmDo(event, success = true){
    let {doing, done, error} = event.detail;
    doing();
    setTimeout(() => {
      if(success){
        done();
      } else {
        error();
      }
    }, 2000)
  }

<Meta
  title="Buttons/Confirm Button"
  component={Button}
  args={args}
/>

<Description markdown={dedent`
# ConfirmButton
## Confirm Button with built-in confirmation.
`}/>

<Canvas>
  <Story
  name="Confirm Button"
  args={args}
  >
  {(args) => ({
    Component: Button,
    props: args,
    on: {
      do : confirmDo,
      done: action("Confirmed")
    },
  })}
  </Story>
</Canvas>
