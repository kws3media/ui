export let args={
  size : "",
  is_active : false
};

export function eraseDo(event, success = true){
  let {doing, done, error} = event.detail;
  doing();
  setTimeout(() => {
    if(success){
      done();
    } else {
      error();
    }
  }, 2000)
}
