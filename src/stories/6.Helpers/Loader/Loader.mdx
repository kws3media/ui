### Usage

##### Import the component:
<Source
  code={ dedent`import { Loader } from '@kws3/ui';`}
/>

##### Import the CSS:
<Source
  code={ dedent`@import '@kws3/ui/styles/Loader.scss';`}
/>

##### SCSS customization variables with default values:
<Source
  language='css'
  code={ dedent`
    $kws-theme-colors: $colors;
    $kws-loader-width: 2rem;
    $kws-loader-height: 2rem;
    $kws-loader-width-medium: ($kws-loader-width * 1.5);
    $kws-loader-height-medium: ($kws-loader-height * 1.5);
    $kws-loader-width-large: ($kws-loader-width * 3);
    $kws-loader-height-large: ($kws-loader-height * 3);
    $kws-loader-color: #ccc;
    $kws-loader-overlay-background: rgba(0,0,0,0.15);
  `}
/>
