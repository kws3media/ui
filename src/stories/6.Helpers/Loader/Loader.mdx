### Usage

##### Import the component:
<Source
  code={ dedent`import { Loader } from '@kws3/ui';`}
/>

##### Import the CSS:
<Source
  code={ dedent`@use '@kws3/ui/styles/Loader.scss';`}
/>

##### SCSS customization variables with default values:
<Source
  language='css'
  code={ dedent`
    @use '@kws3/ui/styles/Loader.scss' with(
      $theme-colors: dv.$colors !default;
      $width: 2rem !default;
      $height: 2rem !default;
      $width-medium: ($width * 1.5) !default;
      $height-medium: ($height * 1.5) !default;
      $width-large: ($width * 3) !default;
      $height-large: ($height * 3) !default;
      $color: #ccc !default;
      $overlay-background: rgba(0, 0, 0, 0.15) !default;
    );
  `}
/>
