### Usage
##### Import the component:
<Source
  code={ dedent`import { Timeline } from '@kws3/ui';`}
/>

##### Import the CSS:
<Source
  code={ dedent`@use '@kws3/ui/styles/Timeline.scss';`}
/>

##### SCSS customization variables with default values:
<Source
  language='css'
  code={ dedent`
    @use '@kws3/ui/styles/Timeline.scss' with(
      $theme-colors: dv.$colors !default;
      $default-color: cv.getVar("grey") !default;
      $default-color-invert: cv.getVar("light") !default;
    );
  `}
/>
