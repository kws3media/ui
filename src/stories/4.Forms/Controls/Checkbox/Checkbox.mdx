### Usage
#### Import the component:
<Source
  code={
    dedent`import { Checkbox } from '@kws3/ui';`
  }
/>

#### Import the SCSS:
<Source
  code={
    dedent`@use '@kws3/ui/styles/CheckRadio.scss';`
  }
/>

#### SCSS customization variables with default values:
<Source
  language="css"
  code={
    dedent`
    @use '@kws3/ui/styles/CheckRadio.scss' with (
      $colors: dv.$colors !default;
      $default-check-color: cv.getVar("text") !default;
      $default-check-color-invert: findColorInvert($default-check-color) !default;
      $default-border-color: #dbdbdb !default;
      $focus-box-shadow-size: cv.getVar("focus-shadow-size") !default;
      $focus-box-shadow-color: hsla(
        cv.getVar("focus-h"),
        cv.getVar("focus-s"),
        cv.getVar("focus-l"),
        cv.getVar("focus-shadow-alpha")
      ) !default;
    );
    `
  }
/>
