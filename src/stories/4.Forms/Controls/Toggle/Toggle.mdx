### Usage
#### Import the component:
<Source
  code={
    dedent`import { Toggle } from '@kws3/ui';`
  }
/>

#### Import the SCSS:
<Source
  code={
    dedent`@use '@kws3/ui/styles/Toggle.scss';`
  }
/>

#### SCSS customization variables with default values:
<Source
  language="css"
  code={
    dedent`
      @use '@kws3/ui/styles/Toggle.scss' with(
        $off-bg-color: cv.getVar("background") !default;
        $handle-bg-color: cv.getVar("white") !default;
        $track-bg-color: cv.getVar("text") !default;
        $track-bg-color-invert: findColorInvert($track-bg-color) !default;
        $border-color: cv.getVar("border") !default;
        $theme-colors: dv.$colors !default;
      );
    `
  }
/>
