### Usage
#### Import the component:
<Source
  code={
    dedent`import { FileUpload } from '@kws3/ui';`
  }
/>

##### Import the SCSS:
<Source
  code={ dedent`@use '@kws3/ui/styles/FileUpload.scss';`}
/>

#### SCSS customization variables with default values:
<Source
  language="css"
  code={
    dedent`
      @use '@kws3/ui/styles/FileUpload.scss with(
        $theme-colors: dv.$colors !default;
        $bg-color: cv.getVar("background-white") !default;
        $border-color: cv.getVar("border") !default;
        $shadow: cv.getVar("shadow-s") !default;
        $radius: cv.getVar("radius") !default;
        $text-color: cv.getVar("text") !default;
        $disabled-bg-color: cv.getVar("white-ter") !default;
        $disabled-border-color: cv.getVar("white-ter") !default;
        $progress-bg-color: cv.getVar("success-light") !default;
        $progress-text-color: cv.getVar("success-dark") !default;
        $progress-anim-easing: linear !default;
      );
    `
  }
/>

#### Example:
<Source
  language="html"
  code={
    dedent`
    <FileUpload
      on:file_chosen={(event) => onFileChosen(event)}
      on:file_uploaded
      on:file_upload_error
      max="1048579"
      allowed={["png", "jpg", "jpeg"]}
      message="Upload new avatar..."
      {disabled}
      classes="" />
    <script>
      import { FileUpload } from "@kws3/ui";
      function onFileChosen(event){
        let {getFile, progress, uploaded, error} = event.detail;
        var {name, key, size, file} = getFile();
        progress(0);
        uploadImage(
          file,
          function(e){
            progress(e.loaded);
          }
        ).then(function(r){
          data = r.response
          uploaded();
        })
        .catch(function(r){
          var user_message = "Failed to upload!";
          if (typeof r != 'undefined' && typeof r.response != 'undefined' && typeof r.response.records != 'undefined'){
              user_message = r.response.records.userMessage;
          }
          error(user_message);
        });
      }
    </script>
  `
  }
/>
