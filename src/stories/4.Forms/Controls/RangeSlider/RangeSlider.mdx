### Usage
#### Import the component:
<Source
  code={
    dedent`import { RangeSlider } from '@kws3/ui';`
  }
/>

#### Import the SCSS:
<Source
  code={
    dedent`@use '@kws3/ui/styles/RangeSlider.scss';`
  }
/>

#### SCSS customization variables with default values:
<Source
  language="css"
  code={
    dedent`
      @use '@kws3/ui/styles/RangeSlider.scss' with(
        $colors: dv.$colors !default;
        $radius: cv.getVar("radius") !default;
        $track-background: cv.getVar("grey-lighter") !default;
        $track-radius: cv.getVar("radius") !default;
        $track-border: 0px solid cv.getVar("grey") !default;
        $track-shadow: 0px 0px 0px cv.getVar("grey") !default;
        $thumb-background: cv.getVar("white") !default;
        $thumb-radius: cv.getVar("radius") !default;
        $thumb-border: 1px solid cv.getVar("grey-light") !default;
        $thumb-shadow: none !default;
        $thumb-to-track-ratio: 2 !default;
        $output-width: 3rem !default;
        $output-background: cv.getVar("grey-dark") !default;
        $output-color: cv.getVar("white") !default;
        $output-radius: cv.getVar("radius") !default;
      );
    `
  }
/>
