### Canvas with background image Example


  <Story name="With Background Image">
    {() => ({
      Component: WithBGImage,
    
    })}
  </Story>


**Example:**

```html

  <Canvas
    bind:image
    active_tool="Pen"
    {width}
    {tools}
    {background_image}
  />
    
<script>
  import { Canvas, Pen, Eraser } from '@kws3/ui';
  
  let let width = "350px";
  let height='250px';
  let background_image = "/images/canvas_bg.png";

  let tools = {Pen, Eraser};

</script>
```
