### ESignature Example

This is example of creating an ESignature/Signature Pad component using Kws3 UI `Canvas` helper  

**Example:**

```html

<Canvas
  bind:image
  bind:is_filled
  {styles}
  {width}
  {height}
  {background_image}
  {readonly}
  {disabled}
  {activeTool}
  {actions}
  {cy}
  {has_controls}
  {control_position} />

<script>
  import { Canvas } from '@kws3/ui';

  export let image, is_filled;
  let styles = {
    border: "1px solid #b5b5b5",
  },
  width = "520px",
  height = "170px",
  background_image = "",
  readonly = false,
  disabled = false,
  actions = ["undo", "redo", "reset"],
  activeTool = "Pen",
  cy = "",
  has_controls = true,
  control_position = "end";

</script>

```
