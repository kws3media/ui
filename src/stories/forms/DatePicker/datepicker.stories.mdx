### Usage
#### Import the component:
<Source
  language="javascript"
  code={
    dedent`import { Datepicker } from '@kws3/ui';`
  }
/>

#### Import the SCSS:
<Source
  language="javascript"
  code={
    dedent`@import 'path/to/node_modules/@kws3/ui/styles/Datepicker.scss';`
  }
/>

#### SCSS customization variables with default values:
<Source
  language="css"
  code={
    dedent`
      $kws-theme-colors: $colors;
      $kws-datepicker-background: $background;
      $kws-datepicker-text: $text;
    `
  }
/>
