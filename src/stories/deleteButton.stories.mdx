import { Story, Preview, Meta, ArgsTable, Source, Description, Canvas  } from '@storybook/addon-docs';
import dedent from 'ts-dedent';
import { DeleteButton as Button } from '@kws3/ui';
import { action } from '@storybook/addon-actions';

export let args={
  color : "primary",
  disabled : false,
  size : "",
  icon_only : false,
  text : "Delete Me",
  icon : "minus-circle",
  doing_icon : "hourglass",
  done_icon : "check",
  doing_text : "Doing",
  done_text : "Deleted"
};

export function eraseDo(event, success = true){
  let {doing, done, error} = event.detail;
  doing();
  setTimeout(() => {
    if(success){
      done();
    } else {
      error();
    }
  }, 2000)
}

<Meta
  title="Buttons/Delete Button"
  component={Button}
  args={args}
  argTypes={{
    color: {
      options: ["primary", "primary", "success", "danger", "warning", "dark", "light"],
      control: { type: 'select' }
    },
    size:{
      options:["small", "medium", "large"],
      control: {type:"select"},
    }
  }}
/>

<Description markdown={dedent`
# DeleteButton
## Delete Button with built-in confirmation.
`}/>

<Canvas>
  <Story
  name="Delete Button"
  args={args}
  >
  {(args) => ({
    Component: Button,
    props: args,
    on: {
      erase : eraseDo,
      erased: action("Erased")
    },
  })}
  </Story>
</Canvas>

<ArgsTable story="Delete Button" />

    dedent`<DeleteButton
